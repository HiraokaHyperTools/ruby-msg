def self.load_chain idx
	buf = idx.read
	type, count = buf.unpack 'v2'
	unless type == 0x0002
		raise 'unknown id2 type 0x%04x' % type
		#return
	end
	id2 = []
	count.times do |i|
		assoc = new buf[8 + SIZE * i, SIZE]
		id2 << assoc
		if assoc.table2 != 0
			id2 += load_chain idx.pst.idx_from_id(assoc.table2)
		end
	end
	id2
end
