def type
        @type ||= begin
                if id & 0x2 == 0
                        :data
                else
                        first_byte, second_byte = read.unpack('CC')
                        if first_byte == 1
                                raise second_byte unless second_byte == 1
                                :data_chain_header
                        elsif first_byte == 2
                                raise second_byte unless second_byte == 0
                                :id2_assoc
                        else
                                raise FormatError, 'unknown first byte for block - %p' % first_byte
                        end
                end
        end
end
